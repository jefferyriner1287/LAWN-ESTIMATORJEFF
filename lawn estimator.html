<script>
    function calculateEstimate() {
        const lawnSize = parseInt(document.getElementById("lawnSize").value);
        const treatmentType = document.getElementById("treatmentType").value;
        const contactInfo = document.getElementById("contactInfo").value;

        // Basic validation
        if (isNaN(lawnSize) || lawnSize <= 0) {
            alert("Please enter a valid lawn size.");
            return;
        }

        if (!contactInfo) {
            alert("Please enter your contact information.");
            return;
        }

        // Base cost calculations
        let baseCost = 0;
        let treatmentName = '';
        let timeline = '';

        if (treatmentType === "aeration") {
            baseCost = Math.max(lawnSize / 1000 * 25, 100); // $25 per 1,000 sq ft, minimum $100
            treatmentName = "Aeration";
            timeline = "4-6 weeks for visible results.";
        } else if (treatmentType === "8-treatments-per-year") {
            baseCost = Math.max(lawnSize / 1000 * 15 * 8, 65); // $15 per 1,000 sq ft per treatment, 8 treatments
            treatmentName = "8 Treatments Per Year (Fertilization + Weed Control)";
            timeline = "2-4 weeks for visible results after each treatment.";
        } else if (treatmentType === "lawn-seeding") {
            baseCost = Math.max(lawnSize / 1000 * 70, 189); // $70 per 1,000 sq ft for fescue
            treatmentName = "Lawn Seeding";
            timeline = "4-6 weeks for visible results.";
        } else if (treatmentType === "one-time-aeration") {
            baseCost = Math.max(lawnSize / 1000 * 25, 100); // Same as Aeration, but for a one-time service
            treatmentName = "One-Time Aeration";
            timeline = "4-6 weeks for visible results.";
        }

        // Display the result on the screen
        document.getElementById("result").innerHTML = `
            <h2>Estimated Cost for ${treatmentName}: $${baseCost.toFixed(2)}</h2>
            <p>Estimated Timeline: ${timeline}</p>
            <p>Contact Info: ${contactInfo}</p>
            <p class="disclaimer">*Price is subject to change depending on the accuracy of the information provided. Additional factors may influence the final cost.</p>
        `;

        // Send the email (using Formspree or another service)
        sendEstimateToEmail(lawnSize, treatmentName, baseCost, timeline, contactInfo);
    }

    function sendEstimateToEmail(lawnSize, treatmentName, baseCost, timeline, contactInfo) {
        // Prepare the data to send via email
        const formData = new FormData();
        formData.append('lawnSize', lawnSize);
        formData.append('treatmentName', treatmentName);
        formData.append('baseCost', baseCost);
        formData.append('timeline', timeline);
        formData.append('contactInfo', contactInfo);

        // Use Formspree or EmailJS API to send the data to your email
        fetch("https://formspree.io/f/your-form-id", {
            method: 'POST',
            body: formData,
        }).then(response => {
            if (response.ok) {
                alert("Estimate sent to you successfully!");
            } else {
                alert("There was an error. Please try again.");
            }
        }).catch(error => {
            alert("There was an error sending the estimate.");
        });
    }
</script>
